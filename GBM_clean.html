GBM exploratory model — transit × amenity capitalisation¶

This notebook reads the state pickled at the end of Code_DiD_V10.ipynb and trains gradient-boosted models (LightGBM + SHAP) to describe how transit accessibility and the local amenity environment jointly relate to residential price/m² across the Montpellier metro.

Run order: restart kernel → Run All. Cell 2 (OSMnx walking distances) is optional — if osmnx isn't installed it skips gracefully and the rest of the pipeline falls back to Euclidean distances from V10.

Outputs: figures are written to DATA_DIR/gbm_figures/.

# V10's distance helper (re-defined locally; functions don't pickle cleanly)
def calc_dist_to_stops(df_input, stops_array):
    coords = df_input[["latitude", "longitude"]].values
    mean_lat = np.radians(coords[:, 0].mean())
    prop_xy = np.column_stack([coords[:, 0] * 111.32,
                               coords[:, 1] * 111.32 * np.cos(mean_lat)])
    stop_xy = np.column_stack([stops_array[:, 0] * 111.32,
                               stops_array[:, 1] * 111.32 * np.cos(mean_lat)])
    return cdist(prop_xy, stop_xy, metric="euclidean").min(axis=1)

# Shared local-planar projection (km), referenced to df_metro centroid
MEAN_LAT_RAD = np.radians(df_metro["latitude"].mean())
def project_planar_km(latlon):
    return np.column_stack([latlon[:, 0] * 111.32,
                            latlon[:, 1] * 111.32 * np.cos(MEAN_LAT_RAD)])

If osmnx is installed, this cell adds dist_*_walk_km columns alongside the Euclidean ones. If not, it skips silently and the model falls back to Euclidean distances. Either choice is defensible for a Discussion-section model; walking distances tighten estimates by ~20% on average.

# Walking-network distances — only if osmnx is available
# ────────────────────────────────────────────────────────────────
WALKING_AVAILABLE = False
try:
    import osmnx as ox
    import networkx as nx

    ox.settings.use_cache = True
    ox.settings.cache_folder = str(DATA_DIR / "osmnx_cache")
    center = (df_metro["latitude"].median(), df_metro["longitude"].median())
    print("Downloading walking graph (cached after first run)…")
    G = ox.graph_from_point(center, dist=10000, network_type="walk", simplify=True)
    print(f"  {len(G.nodes):,} nodes, {len(G.edges):,} edges")

    def snap(latlon):
        return ox.distance.nearest_nodes(G, X=latlon[:, 1], Y=latlon[:, 0])

    nodes_l5   = snap(line5_array)
    nodes_p1   = snap(p1_stops)
    nodes_hist = snap(historic_array)

    for frame in (df_metro, df_p1, df_p2):
        frame["walk_node"] = snap(frame[["latitude", "longitude"]].values)

    def multi_source_min_km(source_nodes, target_nodes, fallback_km):
        sources = set(int(s) for s in source_nodes)
        lengths = nx.multi_source_dijkstra_path_length(
            G, sources=sources, weight="length")
        out = np.array([lengths.get(int(n), np.nan) for n in target_nodes]) / 1000.0
        missing = np.isnan(out)
        if missing.any():
            out[missing] = fallback_km[missing] * 1.3
        return out

    print("Computing network distances…")
    for frame, label in [(df_metro, "metro"),
                         (df_p1,   "Bustram corridor"),
                         (df_p2,   "Phase 2 corridor")]:
        print(f"  {label} ({len(frame):,} obs)")
        frame["dist_line5_walk_km"] = multi_source_min_km(
            nodes_l5, frame["walk_node"].values, frame["dist_line5_km"].values)
        frame["dist_p1_walk_km"]    = multi_source_min_km(
            nodes_p1, frame["walk_node"].values, frame["dist_p1_km"].values)
        frame["dist_hist_walk_km"]  = multi_source_min_km(
            nodes_hist, frame["walk_node"].values, frame["dist_hist_km"].values)

    WALKING_AVAILABLE = True
    ratio = (df_metro["dist_line5_walk_km"] / df_metro["dist_line5_km"])
    print(f"\nDetour ratio (walk/Euclidean) — Line 5:")
    print(f"  median: {ratio.median():.2f}   p10: {ratio.quantile(0.1):.2f}   "
          f"p90: {ratio.quantile(0.9):.2f}")

except ImportError:

    print("  To enable: conda install -n tf-metal -c conda-forge osmnx")
except Exception as e:

def load_geojson_points(path, classify_fn):
    gj = json.load(open(path))
    rows = []
    for feat in gj["features"]:
        if feat["geometry"]["type"] != "Point":
            continue
        lon, lat = feat["geometry"]["coordinates"][:2]
        cat = classify_fn(feat["properties"])
        if cat is not None:
            rows.append({"lat": lat, "lon": lon, "category": cat})
    return pd.DataFrame(rows)

def classify_econ(p):
    s = (p.get("shop") or "").lower()
    if s in LARGE_RETAIL: return "large_retail"
    if s in DAILY_SHOPS:  return "daily_shop"
    if s in SERVICES:     return "services"
    return None

def classify_restau(p):
    a = (p.get("amenity") or "").lower()
    if a in {"cafe", "bar", "pub", "ice_cream"}: return "cafe_bar"
    if a in {"restaurant", "fast_food"}:         return "restaurant"
    return None

amenities = pd.concat([
    load_geojson_points(OSM_ECON,    classify_econ),
    load_geojson_points(OSM_RESTAU,  classify_restau),
    load_geojson_points(OSM_MARKETS, lambda p: "market"),
], ignore_index=True)

# ────────────────────────────────────────────────────────────────
# Buffer counts (300/500/800 m) per category + Shannon diversity
# ────────────────────────────────────────────────────────────────
def add_amenity_features(frame):
    prop_xy = project_planar_km(frame[["latitude", "longitude"]].values)
    for cat in CATEGORIES:
        sub = amenities[amenities["category"] == cat]
        if len(sub) == 0:
            for r in RADII_KM:
                frame[f"n_{cat}_{int(r*1000)}m"] = 0
            continue
        amen_xy = project_planar_km(sub[["lat", "lon"]].values)
        tree = BallTree(amen_xy)
        for r in RADII_KM:
            frame[f"n_{cat}_{int(r*1000)}m"] = tree.query_radius(prop_xy, r=r, count_only=True)

    counts_500 = frame[[f"n_{c}_500m" for c in CATEGORIES]].values.astype(float)
    totals = counts_500.sum(axis=1)
    with np.errstate(divide="ignore", invalid="ignore"):
        p = np.where(counts_500 > 0, counts_500 / np.maximum(totals[:, None], 1), 0.0)
        ent = -np.where(p > 0, p * np.log(p), 0.0).sum(axis=1)
    frame["amenity_diversity_500m"] = np.where(totals > 0, ent, 0.0)
    frame["amenity_total_500m"]     = totals
    return frame

for frame in (df_metro, df_p1, df_p2):
    add_amenity_features(frame)

# Kernel-density features (smoother analogue of buffer counts)

def add_kde_features(frame, bandwidth_km=0.4):
    prop_xy = project_planar_km(frame[["latitude", "longitude"]].values)
    for cat in CATEGORIES:
        sub = amenities[amenities["category"] == cat]
        if len(sub) < 5:
            frame[f"kde_{cat}"] = 0.0
            continue
        amen_xy = project_planar_km(sub[["lat", "lon"]].values)
        kde = KernelDensity(bandwidth=bandwidth_km, kernel="gaussian").fit(amen_xy)
        frame[f"kde_{cat}"] = np.exp(kde.score_samples(prop_xy)) * len(sub)
    return frame

for frame in (df_metro, df_p1, df_p2):

# ────────────────────────────────────────────────────────────────
# Train pooled LightGBM (temporal split: train ≤2022, test ≥2023)
# ────────────────────────────────────────────────────────────────
def fit_lgbm(df, label):
    train = df[df["year"] <= 2022]
    test  = df[df["year"] >= 2023]
    Xtr, ytr = train[FEATURES_NUM + FEATURES_CAT], np.log(train[TARGET])
    Xte, yte = test [FEATURES_NUM + FEATURES_CAT], np.log(test [TARGET])

    m = lgb.LGBMRegressor(
        n_estimators=2000, learning_rate=0.03,
        num_leaves=63 if len(train) > 10_000 else 31,
        min_child_samples=30, subsample=0.8, colsample_bytree=0.8,
        reg_lambda=1.0, random_state=42,
    )
    m.fit(Xtr, ytr, eval_set=[(Xte, yte)],
          categorical_feature=FEATURES_CAT,
          callbacks=[lgb.early_stopping(50), lgb.log_evaluation(0)])
    yp = m.predict(Xte)
    print(f"[{label}] n_train={len(train):,}  n_test={len(test):,}")
    print(f"         OOS R²={r2_score(yte, yp):.3f}  "
          f"OOS MAE(log)={mean_absolute_error(yte, yp):.3f}  "
          f"med |%err|={np.median(np.abs(np.exp(yte)-np.exp(yp))/np.exp(yte))*100:.1f}%")
    return m, Xte, yte

def compute_2d_pdp(model, X, feat_x, feat_y, grid_resolution=25, sample_size=2000, q_lo=0.05, q_hi=0.95):
    """Manual 2D partial dependence: for each (x, y) on a grid,
    overwrite those two columns in a sample and average predictions.
    Returns (x_grid, y_grid, Z) where Z has shape (len(y_grid), len(x_grid))."""
    samp = X.sample(min(sample_size, len(X)), random_state=0).copy()
    x_grid = np.linspace(samp[feat_x].quantile(q_lo), samp[feat_x].quantile(q_hi), grid_resolution)
    y_grid = np.linspace(samp[feat_y].quantile(q_lo), samp[feat_y].quantile(q_hi), grid_resolution)

    Z = np.empty((len(y_grid), len(x_grid)))
    for i, yv in enumerate(y_grid):
        for j, xv in enumerate(x_grid):
            samp[feat_x] = xv
            samp[feat_y] = yv
            Z[i, j] = model.predict(samp).mean()
    return x_grid, y_grid, Z
